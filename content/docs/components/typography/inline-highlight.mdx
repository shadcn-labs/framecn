## Usage

```tsx
import { InlineHighlight } from "@/components/ui/inline-highlight";
```

```tsx
<InlineHighlight
  text="text"
  highlightWords={0}
  fromColor="fromColor"
  toColor="toColor"
/>
```

## API Reference

### InlineHighlight

| Prop             | Type       | Default     |
| ---------------- | ---------- | ----------- |
| `text`           | `string`   | `-`         |
| `highlightWords` | `number[]` | `-`         |
| `fromColor`      | `string`   | `"#666666"` |
| `toColor`        | `string`   | `"#3b82f6"` |
